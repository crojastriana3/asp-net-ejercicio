CREATE DATABASE ELECTIVAUNO

USE ELECTIVAUNO


CREATE TABLE EMPRESA(
 nit INT PRIMARY KEY NOT NULL,
 direccion VARCHAR(40),
 telefono varchar(10)
);

SELECT * FROM EMPRESA;

CREATE TABLE DEPARTAMENTOS(
	codigo INT PRIMARY KEY NOT NULL,
	nombre VARCHAR(40),
	funcion VARCHAR(40)
);

CREATE TABLE EMPLEADOS(
	codigo INT PRIMARY KEY NOT NULL,
	tipoDocumeneto VARCHAR(20),
	numeroDocumento INT,
	nombre VARCHAR(50),
	cargo VARCHAR(30),
	sueldo MONEY
);
